namespace = ho_on_death

# 近しい人が死んだら誰が死んだのか保存
ho_on_death.0001 = {
	hidden = yes
	scope = character

	immediate = {
		every_close_family_member = {
			set_variable = {
				name = death_close_person
				value = root
				years = 5
			}
		}

		every_spouse = {
			limit = {
				NOR = {
					has_relation_friend = root
					has_relation_rival = root
					has_relation_lover = root
					is_close_family_of = root
				}
			}
			set_variable = {
				name = death_close_person
				value = root
				years = 5
			}
		}
	}
}
